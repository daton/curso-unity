               using UnityEngine;
using System.Collections;

public class MovimientoMouse : MonoBehaviour {
    public float rapidez = 2.0f;
	// Use this for initialization
    float rotarX=0;

        public enum RotacionEjes
    {
        MouseXY=0,
        MouseX=1,
        MouseY=2
    }

    public RotacionEjes ejes = RotacionEjes.MouseX;
	void Start () {
       
	}
	
	// Update is called once per frame
	void Update () {
        // transform.Rotate(0, rapidez, 0);
        if (ejes == RotacionEjes.MouseX)
        {
            transform.Rotate(0, Input.GetAxis("Mouse X") * 2, 0);
        }
        else if (ejes == RotacionEjes.MouseY)
        {
            rotarX -= Input.GetAxis("Mouse Y") * 2;
            Debug.Log("miuan");
            rotarX = Mathf.Clamp(rotarX, -40, 40);
            float rotarY = transform.localEulerAngles.y;
            transform.localEulerAngles = new Vector3(rotarX, rotarY, 0);
        }
        else
        {
            rotarX -= Input.GetAxis("Mouse Y") * rapidez;
            rotarX = Mathf.Clamp(rotarX, -45, 60);
            float deltax = Input.GetAxis("Mouse X") * rapidez;
            float rotarY = transform.localEulerAngles.y + deltax;
            transform.localEulerAngles = new Vector3(rotarX, rotarY, 0);  
        }
    }
}
